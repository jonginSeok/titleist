1. Init
$(selector).grid('init', uurl, title, model, option, event)
- 기본적인 Grid를 만드는 함수.
Returns
- Grid Object
Parameters
- uurl (grid 데이타를 만들 xml 파일의 경로)
- title (그리드 Header 이름)
- model (그리드 칼럼 속성을 정의)
- option (필요한 그리드 속성 정의)
- event (필요한 그리드의 이벤트 정의)
Example
//Grid title 
var gTitle = ['Inv No','Date', 'Client', 'Amount','Tax','Total','Notes'];
//Grid Model 
var gModel = [
	{name:'id',index:'id', width:60, sorttype:'int',summaryType:'count', summaryTpl : '({0}) total'},
	{name:'invdate',index:'invdate', width:90, sorttype:"date"},
	{name:'name',index:'name', width:100},
	{name:'amount',index:'amount', width:80, align:'right',sorttype:"float",formatter:"number", 
			summaryType:"sum"},
	{name:'tax',index:'tax', width:80, align:'right',sorttype:'number',formatter:'number',summaryType:'sum'},		
	{name:'total',index:'total', width:80,align:'right',sorttype:'number',formatter:'number', summaryType:'sum'},
	{name:'note',index:'note', width:150, sortable:false,editable:true}		
];

var options = 	{
	//옵션에 multiselect:true 부분을 추가하면 그리드 목록 전체선택/전체 해제가 가능하다. 
	multiselect:true,	
	pager: '#pager1', 	
	rowNum:3,
   	rowList:[3,6,9,12,15],
   	xmlReader: {
		root : "data",
		row: "list", 
		page: "data>page", 			//
	    total: "data>total", 		//
	    records : "data>records", 	//
		repeatitems: false,
		id: "id"
		}
	
	//옵션에 groupFlag: true 부분을 추가하면 소계/합계를 보여줄수 있다. 
	groupFlag: true,
   	groupView : {
   		//소계와 관련된 옵션
   		groupField : ['name'],
   		groupColumnShow : [true],
   		groupText : ['b>{0}'],
   		groupCollapse : false,
		groupOrder: ['asc'],
		groupSummary : [true],
		showSummaryOnHide: true,
		groupDataSorted : true
   	},
   	//합계로우 보여주기
    footerrow: true,
    userDataOnFooter: true
};

var events = {loadComplete:fn_loadComplete1, ondblClickRow:fn_ondblClickRow1, 
				onRightClickRow:fn_onRightClickRow1};

//1. 기본적인 Grid을 만든다. : 화면 로딩시 바로 데이타 조회
$k1("#list1").grid('init','../jsp/grid_data2.jsp', gTitle, gModel, options, events);

22. dataToString
$(selector).grid('dataToString', rowid, editYN)
- 해당 Row의 칼럼별 데이타를 String를 얻는 함수.
Returns
- 데이타 String
Parameters
- rowid (선택한 Row 아이디)
- editYN (edit 상태인지 여부)
Example
// row을 선택시에 edit한 상태로 변경한다. 
function gfnCardSelectRow (id) {
	if(id && id!==lastsel2){
		if(!$k1.util.isNull(lastsel2)) {
			uptData = $k1('#tbl_card').grid('dataToString', lastsel2, true);
			
			isSame = (orgData == uptData)? true:false;
		}
		//데이터가 변경되었을 경우에만 업데이트한 데이타를 구한다. 
		if(!isSame) {
			updateData = $k1('#tbl_card').grid('editDataRow', lastsel2);  
		}
		//위에서 구한 업데이트한 데이타를  restoreRow 한 후에 setData한다. 
		$k1('#tbl_card').restoreRow(lastsel2, {"afterrestorefunc" : function( response ) {
				if(!isSame) {
					$k1("#tbl_card").jqGrid('setRowData',lastsel2, updateData);
					//makeJSON();
					updateData = [], isSame=true;
					$k1('#tbl_card').grid('setCombo', lastsel2); 
				}
			}}
		); 
		
		lastsel2=id; 
	}

	//original data 얻기
	orgData = $k1('#tbl_card').grid('dataToString', id, false);
	$k1('#tbl_card').editRow(id,true);

}

25. setComboData

Returns
-
Parameters
- rowid (선택한 Row 아이디)

Example
var $target = $(this);
var colModel = $target.grid('getGridParam', 'colModel');
$.each(colModel, function(index, col) {
	if (col.edittype === 'select') {
		if ($.util.isNull(rowid)) {
			//전체 Row의 combo를 세팅
			$.each($target.jqGrid('getDataIDs'), function(index, id) {
				$target.grid('setComboData', id, col);
			});
		} else {
			//특정 Row의 combo를 세팅
			$target.grid('setComboData', rowid, col);
		}
	}
});
